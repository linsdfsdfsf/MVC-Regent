CREATE USER "MSP_RM"  PROFILE "DEFAULT" 
    IDENTIFIED BY "msp_rm" DEFAULT TABLESPACE "MSP_RM_DAT" 
    TEMPORARY TABLESPACE "TEMP" 
    ACCOUNT UNLOCK;
GRANT DELETE ANY TABLE TO "MSP_RM";
GRANT EXECUTE ANY PROCEDURE TO "MSP_RM";
GRANT CREATE VIEW TO "MSP_RM";
GRANT CREATE TRRIGER TO "MSP_RM";
GRANT INSERT ANY TABLE TO "MSP_RM";
GRANT SELECT ANY DICTIONARY TO "MSP_RM";
GRANT DEBUG  CONNECT SESSION TO "MSP_RM";
GRANT SELECT ANY SEQUENCE TO "MSP_RM";
GRANT SELECT ANY TABLE TO "MSP_RM";
GRANT UNLIMITED TABLESPACE TO "MSP_RM";
GRANT UPDATE ANY TABLE TO "MSP_RM";
GRANT "CONNECT" TO "MSP_RM";
GRANT "RESOURCE" TO "MSP_RM";
REVOKE UNLIMITED TABLESPACE 
    FROM "MSP_RM";
ALTER USER "MSP_RM";
    QUOTA UNLIMITED 
    ON "MSP_RM_DAT" 
    QUOTA UNLIMITED 
    ON "MSP_RM_IDX";
