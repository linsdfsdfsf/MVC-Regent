CREATE USER "MSP_SYS"  PROFILE "DEFAULT" 
    IDENTIFIED BY "msp_sys" DEFAULT TABLESPACE "MSP_SYS_DAT" 
    TEMPORARY TABLESPACE "TEMP" 
    ACCOUNT UNLOCK;
GRANT DELETE ANY TABLE TO "MSP_SYS";
GRANT EXECUTE ANY PROCEDURE TO "MSP_SYS";
GRANT CREATE VIEW TO "MSP_SYS";
GRANT CREATE TRRIGER TO "MSP_SYS";
GRANT INSERT ANY TABLE TO "MSP_SYS";
GRANT SELECT ANY DICTIONARY TO "MSP_SYS";
GRANT DEBUG  CONNECT SESSION TO "MSP_SYS";
GRANT SELECT ANY SEQUENCE TO "MSP_SYS";
GRANT SELECT ANY TABLE TO "MSP_SYS";
GRANT UNLIMITED TABLESPACE TO "MSP_SYS";
GRANT UPDATE ANY TABLE TO "MSP_SYS";
GRANT "CONNECT" TO "MSP_SYS";
GRANT "RESOURCE" TO "MSP_SYS";
REVOKE UNLIMITED TABLESPACE 
    FROM "MSP_SYS";
ALTER USER "MSP_SYS";
    QUOTA UNLIMITED 
    ON "MSP_SYS_DAT" 
    QUOTA UNLIMITED 
    ON "MSP_SYS_IDX";
